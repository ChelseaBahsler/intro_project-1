  def display_next_row(content_inside)
    content_outer = ""
    count = 0

    until count == @cards.length()
        content_outer = "<div class='tile is-ancestor'>#{content_inside}</div>"
      else
        count += 1
      end
      content_outer.html_safe
    end
  end
